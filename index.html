function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const data = sheet.getDataRange().getValues(); // All data const username = e.parameter.username; const password = e.parameter.password; const ceu = e.parameter.ceu; for (let i = 1; i < data.length; i++) { if ( data[i][0] === username && data[i][1] === password && data[i][2] === ceu ) { return ContentService.createTextOutput("success"); } } return ContentService.createTextOutput("error"); }
